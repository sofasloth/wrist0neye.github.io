#### Lazy Loading
```js
// 데이터 자체를 필요할 때만 로드
const loadedData = new Map() // 처음엔 비어있음

async function getCellData(row, col) {
  const key = `${row},${col}`
  
  if (!loadedData.has(key)) {
    // 처음 접근시에만 데이터 로드
    const data = await fetchCellData(row, col)
    loadedData.set(key, data)
  }
  
  return loadedData.get(key)
}
```
